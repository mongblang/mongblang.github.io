```python
from itertools import permutations

def solution(numbers):
    answer = 0
    all_nums = set()
    
    for i in range(1, len(numbers)+1):
        for j in permutations(numbers, i):
            all_nums.add(int(''.join(j)))
    
    for i in all_nums:
        if i < 2:
            continue
        if i == 2:
            answer += 1
            continue
        if i % 2 == 0:
            continue
        
        for j in range(3, int(i**0.5)+1, 2):
            if i % j == 0:
                break
        else:
            answer += 1
    
    return answer
```
