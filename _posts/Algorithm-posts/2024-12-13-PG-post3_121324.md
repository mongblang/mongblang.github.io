def solution(participant, completion):
    dict = {}
    
    for i in participant:
        if i in dict:
            dict[i] += 1
        else:
            dict[i] = 1
    
    for c in completion:
        dict[c] -= 1

    names = dict.keys()
    for i in names:
        if dict[i] == 1: return i
