def solution(lines):
    num = []
    for start, end in lines:
        num.append(start)
        num.append(end)
    num.sort()
    
    length = 0
    for i in range(len(num)-1): # end x 
        s = num[i]
        e = num[i+1]
        mid = (s+e)/2
    
        cnt = 0
        for start, end in lines:
            if start <= mid < end:
                cnt += 1
        if cnt >= 2:
            length += e - s
        
    return length
