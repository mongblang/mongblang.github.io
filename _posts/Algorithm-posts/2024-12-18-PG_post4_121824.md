def solution(babbling):
    cnt = 0
    words = ["aya", "ye", "woo", "ma"]
    
    for i in babbling:
        if "ayaaya" in i or "yeye" in i or "woowoo" in i or "mama" in i:
            continue

        for w in words:
            i = i.replace(w," ")

        if i.replace(" ","") == "":
            cnt += 1
            
    return cnt
