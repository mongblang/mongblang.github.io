```python
from collections import Counter

def solution(X, Y):
    cnt_x, cnt_y = Counter(str(X)), Counter(str(Y))
    pair = cnt_x & cnt_y
    
    num = []
    for i,v in pair.items():
        num.extend(i*v)
    
    result = sorted(num, reverse=True)
    
    if not result:
        return "-1"
    elif result[0] == "0":
        return "0"
    else:
        return ''.join(result)
```
