from functools import cmp_to_key

def solution(numbers):

    def compare(x,y):
        if x+y > y+x :
            return -1
        elif x+y == y+x:
            return 0
        else:
            return 1
    
    numbers = [str(i) for i in numbers]
    return str(int(''.join(sorted(numbers, key = cmp_to_key(compare)))))
