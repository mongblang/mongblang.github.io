def solution(schedules, timelogs, startday):
    madeit = 0 
    
    for i in range(len(timelogs)): 
        cnt = 0 
        for j in range(len(timelogs[i])): 
            days = (startday + j) % 7
            if days == 0:
                days = 7 

            if days <= 5: 
                h = schedules[i] // 100     
                m = schedules[i] % 100 + 10 

                if m >= 60:
                    h += 1
                    m -= 60 
                
                late = h*100 + m 
                if timelogs[i][j] <= late:
                    cnt += 1
                else:
                    cnt = 0 
                    break
                
                if cnt == 5: 
                    madeit += 1
    return madeit
