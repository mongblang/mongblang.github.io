def solution(progresses, speeds):
    queue = deque()

        work_left = 100-progresses[i]
        day = -(work_left * (-1) // speeds[i])
        queue.append(day)

    answer = []

    while queue: 
        first_item = queue.popleft() 
        cnt = 1 

        while queue and queue[0] <= first_item:

        answer.append(cnt)
    
    return answer

def solution(progresses, speeds):
    Q=[]
    for p, s in zip(progresses, speeds):
        if len(Q)==0 or Q[-1][0]<-((p-100)//s):
