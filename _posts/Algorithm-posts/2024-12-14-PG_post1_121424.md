def solution(s, n):
    ans = []
    for i in s:
        if i == ' ':
            ans.append(' ')
        elif i.isupper():
            ans.append(chr((ord(i) - 65 + n)% 26 + 65))
        elif i.islower():
            ans.append(chr((ord(i) - 97 + n)%26 + 97))
    return ''.join(ans)
